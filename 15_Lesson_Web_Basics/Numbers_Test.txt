def generate_numbers(start_prefix, total_digits):
    """
    Generate numbers with a fixed prefix.

    Example:
    Prefix: 77
    Total digits: 9
    Output:
    770000000 -> 779999999
    """

    variable_digits = total_digits - len(str(start_prefix))

    start = int(f"{start_prefix}" + "0" * variable_digits)
    end = int(f"{start_prefix}" + "9" * variable_digits)

    for number in range(start, end + 1):
        print(number)

# Run the program
generate_numbers(77, 9)

#include <iostream>
#include <string>
#include <cmath>

void generateNumbers(int prefix, int totalDigits) {

    std::string prefixStr = std::to_string(prefix);

    int variableDigits = totalDigits - prefixStr.length();

    long long start =
        std::stoll(prefixStr + std::string(variableDigits, '0'));

    long long end =
        std::stoll(prefixStr + std::string(variableDigits, '9'));

    for (long long number = start; number <= end; number++) {
        std::cout << number << std::endl;
    }
}

int main() {

    generateNumbers(77, 9);

    return 0;
}
